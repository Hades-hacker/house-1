// Complete Node.js Payment Backend
const express = require('express');
const mongoose = require('mongoose');
const crypto = require('crypto');
const app = express();

app.use(express.json());

// MongoDB Schema
const transactionSchema = new mongoose.Schema({
    transactionId: String,
    method: String,
    amount: Number,
    currency: { type: String, default: 'KES' },
    status: String,
    customerEmail: String,
    propertyId: String,
    guests: Number,
    mpesaCode: String,
    cardLast4: String,
    bankRef: String,
    createdAt: { type: Date, default: Date.now }
});

const Transaction = mongoose.model('Transaction', transactionSchema);

// M-PESA Payment Handler
app.post('/api/payments/mpesa', async (req, res) => {
    try {
        const { phone, amount, customerEmail, propertyId, guests } = req.body;
        const mpesaCode = 'MPESA-' + crypto.randomBytes(3).toString('hex').toUpperCase();
        
        const transaction = new Transaction({
            transactionId: mpesaCode,
            method: 'M-PESA',
            amount,
            status: 'COMPLETED',
            customerEmail,
            propertyId,
            guests,
            mpesaCode
        });
        
        await transaction.save();
        
        res.json({ 
            success: true, 
            mpesaCode,
            status: 'COMPLETED',
            amount,
            message: 'STK push sent to phone'
        });
    } catch (error) {
        res.status(500).json({ error: error.message });
    }
});

// Card Payment Handler (Visa/Mastercard/Amex/Debit)
app.post('/api/payments/card', async (req, res) => {
    const { cardNumber, amount, customerEmail, propertyId, guests, method } = req.body;
    const last4 = cardNumber.slice(-4);
    const authCode = 'AUTH-' + crypto.randomBytes(2).toString('hex').toUpperCase();
    
    const transaction = new Transaction({
        transactionId: authCode,
        method,
        amount,
        status: 'COMPLETED',
        customerEmail,
        propertyId,
        guests,
        cardLast4: last4
    });
    
    await transaction.save();
    
    res.json({ success: true, authCode, last4, status: 'APPROVED' });
});

// PayPal Handler
app.post('/api/payments/paypal', async (req, res) => {
    const { amount, customerEmail, propertyId, guests } = req.body;
    const paypalRef = 'PAY-' + crypto.randomBytes(4).toString('hex').toUpperCase();
    
    const transaction = new Transaction({
        transactionId: paypalRef,
        method: 'PayPal',
        amount,
        status: 'COMPLETED',
        customerEmail,
        propertyId,
        guests
    });
    
    await transaction.save();
    res.json({ success: true, paypalRef, status: 'COMPLETED' });
});

// Bank Transfer Handler
app.post('/api/payments/bank', async (req, res) => {
    const { amount, customerEmail, propertyId, guests } = req.body;
    const bankRef = 'TRF-' + crypto.randomBytes(3).toString('hex').toUpperCase();
    
    const transaction = new Transaction({
        transactionId: bankRef,
        method: 'Bank Transfer',
        amount,
        status: 'PENDING',
        customerEmail,
        propertyId,
        guests,
        bankRef
    });
    
    await transaction.save();
    res.json({ success: true, bankRef, status: 'PENDING' });
});

// Get all transactions
app.get('/api/transactions', async (req, res) => {
    const transactions = await Transaction.find().sort({ createdAt: -1 });
    res.json(transactions);
});

// M-PESA Callback Webhook
app.post('/mpesa/callback', (req, res) => {
    console.log('M-PESA callback:', req.body);
    res.json({ ResultCode: 0, ResultDesc: 'Success' });
});

app.listen(3000, () => {
    console.log('✅ BHAST HOUSE backend running on port 3000');
});
